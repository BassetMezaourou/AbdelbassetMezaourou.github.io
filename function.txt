// Program 9: Factorial and Permutation Calculation
int fact(int N)
{
	int x,r;
	x=1;
	for (r=2;r<=N;r++)
	x=x*r;
	return x;
}
 
int A(int a, int b)
{
	int s;
	s=fact(a)/fact(a-b);
	return s;
}

void revers(int *t, int n) {
    int i, j, c;
    i = 0;
    j = n - 1;
    while (i < j) {
        c = t[i];
        t[i] = t[j];
        t[j] = c;
        i++;
        j--;
    }
}

void sorting_array(int *t, int n) {
    int i, j, c;
    for (i = 0; i < n - 1; i++) 
  {
        for (j = 0; j < n - i - 1; j++) {
            if (t[j] > t[j + 1]) {
                c = t[j];
                t[j] = t[j + 1];
                t[j + 1] = c;
            }
        }
    }
}

} Student;

    }
}

// Function to check if a student is admitted based on average grade
bool is_admitted(Student s) {
    return (s.AG >= 10);
}

// Function to sort students by grade (descending order)
void sort_students_by_grade(Student *s, int n) {
    int i;
    bool swap = true;
    Student temp;
    
    while (swap) {
        swap = false;
        for (i = 0; i < (n - 1); i++) {
            if (s[i].AG < s[i + 1].AG) {
                temp = s[i];
                s[i] = s[i + 1];
                s[i + 1] = temp;
                swap = true;
            }
        }
    }
}

// Function to search students by last name in file
void search_student_file(char path[100], char last_name[15]) {
    FILE *file = fopen(path, "r");
    if (file) {
        char ln[15], fn[15];
        int age;
        float ag;
        int find = 0;
        
        while (fscanf(file, "%14[^_]_%14[^_]_Age=%d_Average=%f\n", 
                     ln, fn, &age, &ag) == 4) {
            if (strcmp(ln, last_name) == 0) {
                if (find == 0) {
                    printf("Students found with last name %s:\n", last_name);
                }
                printf("\nStudent %d:\n", find + 1);
                printf("Last Name: %s\n", ln);
                printf("First Name: %s\n", fn);
                printf("Age: %d\n", age);
                printf("Average Grades: %.2f\n", ag);
                find++;
            }
        }
        if (!find) {
            printf("No students with last name %s found.\n", last_name);
        }
        fclose(file);
    } else {
        printf("[ %s ] File does not exist\n", path);
    }
}

//ReverseLL: qui permet d’inverser la liste

list* ReverseLL(list* head) {
    if (head == NULL || getnextp(head) == NULL) {
        return head;  
    }
    
    list* prev = NULL;
    list* p = head;
    list* next = NULL;
    
    while (p != NULL) {
        next = getnextp(p);
        setnextp(p, prev);
        prev = p;
        p = next;
    }
    
    return prev;
}

list* RemoveLL(list* head, int v) {
    if(head == NULL) return NULL;
    list* q = FindLL(head, v);
    if(q == NULL) return head;
    if(q == head) {
        list* p = getnext(head);
        free(head);
        return p;
    }
    list* prev = head;
    while(getnext(prev) != q) prev = getnext(prev);
    setnext(prev, getnext(q));
    free(q);
    return head;
}
